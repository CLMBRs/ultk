 24@dataclass(frozen=True)
 25class Rule:
 26    """Basic class for a grammar rule.  Grammar rules in ULTK correspond
 27    to functions.  One can think of a grammar as generating complex functions from
 28    more basic ones.
 29
 30    Attributes:
 31        lhs: left-hand side of the rule (can be anything)
 32            conceptually, the output type of a function
 33        rhs: right-hand side; assumed to be an iterable
 34            conceptually, a list of types of inputs
 35        func: a callable, the function to be computed when a node with this rule is executed
 36        name: name of the function
 37        weight: a relative weight to assign to this rule
 38            when added to a grammar, all rules with the same LHS will be weighted together
 39    """
 40
 41    name: str
 42    lhs: Any
 43    rhs: Sequence | None
 44    func: Callable = lambda *args: None
 45    weight: float = 1.0
 46
 47    def is_terminal(self) -> bool:
 48        """Whether this is a terminal rule.  In our framework, this means that RHS is empty,
 49        i.e. there are no arguments to the function.
 50        """
 51        return self.rhs is None
 52
 53    def __str__(self) -> str:
 54        out_str = f"{str(self.lhs)} -> {self.name}"
 55        if self.rhs is not None:
 56            out_str += f"({', '.join(str(typ) for typ in self.rhs)})"
 57        return out_str

 59    @classmethod
 60    def from_callable(cls, func: Callable) -> "Rule":
 61        """Create a Rule from the type-annotations of a function.
 62
 63        For example, given the following method definition:
 64        ```python
 65        def _and(p1: bool, p2: bool) -> bool:
 66            return p1 and p2
 67        ```
 68
 69        This class method will return the Rule:
 70        ```
 71        Rule(name="_and", lhs=bool, rhs=(bool, bool), func=_and)
 72        ```
 73
 74        There are two special kwargs that can be used in the function definition:
 75        - `weight`: a float, which will be used as the weight of the rule
 76        - `name`: a string, which will be used as the name of the rule, if you want it to be different than the name of the method
 77        """
 78        annotations = inspect.signature(func)
 79        if annotations.return_annotation is inspect.Signature.empty:
 80            raise ValueError(
 81                f"Function {func} must have a return annotation to be used as a Rule."
 82            )
 83        # dict so that deletion is possible
 84        weight: float = 1.0
 85        args = dict(annotations.parameters)
 86        if "weight" in args:
 87            # assign weight as the default value of weight kwarg
 88            weight = float(args["weight"].default)
 89            # delete because weight is a special term, not part of RHS like other params
 90            del args["weight"]
 91        # allow custon names too
 92        rule_name = func.__name__
 93        if "name" in args:
 94            rule_name = args["name"].default
 95            del args["name"]
 96        # parameters = {'name': Parameter} ordereddict, so we want the values
 97        # each value is a Paramter, with .annotation being the actual annotation
 98        rhs: tuple[Any, ...] | None = tuple(arg.annotation for arg in args.values())
 99        # if one type annotation, a type of Referent, treat this as a terminal, no children = None RHS
100        # TODO: make this more general?
101        if rhs and len(rhs) == 1 and issubclass(rhs[0], Referent):
102            rhs = None
103        return cls(
104            name=rule_name,
105            lhs=annotations.return_annotation,
106            rhs=rhs,
107            func=func,
108            weight=weight,
109        )

114class GrammaticalExpression(Expression[T]):
115    """A GrammaticalExpression has been built up from a Grammar by applying a sequence of Rules.
116    Crucially, it is _callable_, using the functions corresponding to each rule.
117
118    A GrammaticalExpression, when called, takes in a Referent.  Because of this, a Meaning can
119    be generated by specifying a Universe (which contains Referents).
120
121    Attributes:
122        rule_name: name of the top-most function
123        func: the function
124        children: child expressions (possibly empty)
125    """
126
127    rule_name: str
128    func: Callable
129    children: tuple | None
130    term_expression: str = ""
131
132    def __post_init__(self):
133        if not self.term_expression:
134            self.term_expression = str(self)
135
136    def yield_string(self) -> str:
137        """Get the 'yield' string of this term, i.e. the concatenation
138        of the leaf nodes.
139
140        This is useful for thinking of a `Grammar` as generating derivation trees for
141        an underlying CFG.  This method will then generate the strings generated by
142        the corresponding CFG.
143        """
144        if self.children is None:
145            return str(self)
146        return "".join(child.yield_string() for child in self.children)
147
148    def evaluate(self, universe: Universe) -> Meaning:
149        # NB: important to use `not self.meaning` and not `self.meaning is None` because of how
150        # Expression.__init__ initializes an "empty" meaning if `None` is passed
151        if not self.meaning:
152            self.meaning = Meaning(
153                FrozenDict(
154                    {referent: self(referent) for referent in universe.referents}
155                ),
156                universe,
157            )
158        return self.meaning
159
160    def add_child(self, child) -> None:
161        if self.children is None:
162            self.children = tuple([child])
163        else:
164            self.children = self.children + (child,)
165
166    def to_dict(self) -> dict:
167        the_dict = super().to_dict()
168        the_dict["term_expression"] = self.term_expression
169        the_dict["rule_name"] = self.rule_name
170        the_dict["length"] = len(self)
171        if self.children:
172            the_dict["children"] = tuple(child.to_dict() for child in self.children)
173        return the_dict
174
175    @classmethod
176    def from_dict(cls, the_dict: dict, grammar: "Grammar") -> "GrammaticalExpression":
177        children = the_dict.get("children")
178        if children:
179            children = tuple(cls.from_dict(child, grammar) for child in children)
180        return cls(
181            rule_name=the_dict["rule_name"],
182            func=grammar._rules_by_name[the_dict["rule_name"]].func,
183            children=children,
184            term_expression=the_dict["term_expression"],
185            meaning=the_dict["meaning"],
186        )
187
188    def __call__(self, *args):
189        if self.children is None:
190            return self.func(*args)
191        return self.func(*(child(*args) for child in self.children))
192
193    def __len__(self):
194        length = 1
195        if self.children is not None:
196            length += sum(len(child) for child in self.children)
197        return length
198
199    def __lt__(self, other) -> bool:
200        if not isinstance(other, GrammaticalExpression):
201            raise TypeError(f"Cannot compare GrammaticalExpression with {type(other)}.")
202        children = self.children or tuple([])
203        other_children = other.children or tuple([])
204        return (self.rule_name, self.form, self.func, children) < (
205            other.rule_name,
206            other.form,
207            other.func,
208            other_children,
209        )
210
211    def __str__(self):
212        out_str = self.rule_name
213        if self.children is not None:
214            out_str += f"({', '.join(str(child) for child in self.children)})"
215        return out_str
216
217    def __repr__(self):
218        return f"GrammaticalExpression({self.form}, {self.rule_name}, {self.children}, {self.term_expression}, {self.meaning})"

237class Grammar:
238    """At its core, a Grammar is a set of Rules with methods for generating GrammaticalExpressions."""
239
240    def __init__(self, start: Any):
241        # _rules: nonterminals -> list of rules
242        self._rules: dict[Any, list[Rule]] = defaultdict(list)
243        # name -> rule, for fast lookup in parsing
244        self._rules_by_name: dict[str, Rule] = {}
245        self._start = start
246
247    def add_rule(self, rule: Rule):
248        self._rules[rule.lhs].append(rule)
249        if rule.name in self._rules_by_name:
250            raise ValueError(
251                f"Rules of a grammar must have unique names. This grammar already has a rule named {rule.name}."
252            )
253        self._rules_by_name[rule.name] = rule
254
255    def parse(
256        self,
257        expression: str,
258        opener: str = "(",
259        closer: str = ")",
260        delimiter: str = ",",
261    ) -> GrammaticalExpression:
262        """Parse a string representation of an expression of a grammar.
263        Note that this is not a general-purpose parsing algorithm.  We assume that the strings are of the form
264            parent_name(child1_name, ..., childn_name)
265        where parent_name is the name of a rule of this grammar that has a length-n RHS, and that
266        childi_name is the name of a rule for each child i.
267
268        Args:
269            expression: string in the above format
270
271        Returns:
272            the corresponding GrammaticalExpression
273        """
274        # see nltk.tree.Tree.fromstring for inspiration
275        # tokenize string roughly by splitting at open brackets, close brackets, and delimiters
276        open_re, close_re, delimit_re = (
277            re.escape(opener),
278            re.escape(closer),
279            re.escape(delimiter),
280        )
281        name_pattern = f"[^{open_re}{close_re}{delimit_re}]+"
282        token_regex = re.compile(
283            rf"{name_pattern}{open_re}|{name_pattern}|{delimit_re}\s*|{close_re}"
284        )
285
286        # stack to store the tree being built
287        stack: list[GrammaticalExpression] = []
288
289        for match in token_regex.finditer(expression):
290            # strip trailing whitespace if needed
291            token = match.group().strip()
292            # start a new expression
293            if token[-1] == opener:
294                name = token[:-1]
295                stack.append(
296                    GrammaticalExpression(
297                        rule_name=name,
298                        func=self._rules_by_name[name].func,
299                        children=tuple(),
300                    )
301                )
302            # finish an expression
303            elif token == delimiter or token == closer:
304                # finished a child expression
305                # TODO: are there edge cases that distinguish delimiter from closer?
306                child = stack.pop()
307                stack[-1].add_child(child)
308            else:
309                # primitive, no children, just look up
310                stack.append(
311                    GrammaticalExpression(
312                        rule_name=token,
313                        func=self._rules_by_name[token].func,
314                        children=None,
315                    )
316                )
317        if len(stack) != 1:
318            raise ValueError("Could not parse string {expression}")
319        return stack[0]
320
321    def generate(self, lhs: Any = None) -> GrammaticalExpression:
322        """Generate an expression from a given lhs."""
323        if lhs is None:
324            lhs = self._start
325        rules = self._rules[lhs]
326        the_rule = random.choices(rules, weights=[rule.weight for rule in rules], k=1)[
327            0
328        ]
329        children = (
330            None
331            if the_rule.rhs is None
332            else tuple([self.generate(child_lhs) for child_lhs in the_rule.rhs])
333        )
334        # if the rule is terminal, rhs will be empty, so no recursive calls to generate will be made in this comprehension
335        return GrammaticalExpression(
336            rule_name=the_rule.name, func=the_rule.func, children=children
337        )

339    def enumerate(
340        self,
341        depth: int = 8,
342        lhs: Any = None,
343        uniqueness_args: UniquenessArgs | None = None,
344    ) -> Generator[GrammaticalExpression, None, None]:
345        """Enumerate all expressions from the grammar up to a given depth from a given LHS.
346        This method also can update a specified dictionary to store only _unique_ expressions, with
347        a user-specified criterion of uniqueness.
348
349        Args:
350            depth: how deep the trees should be
351            lhs: left hand side to start from; defaults to the grammar's start symbol
352            uniqueness_args: a dictionary specifying the parameters for uniqueness:
353                unique_dict: a dictionary in which to store unique Expressions
354                key: a function used to evaluate uniqueness
355                compare_func: a comparison function, used to decide which Expression to add to the dict
356                    new Expressions will be added as values to `unique_dict` only if they are _minimal_
357                    among those sharing the same key (by `unique_key`) according to this func
358
359        Yields:
360            all GrammaticalExpressions up to depth
361        """
362        if lhs is None:
363            lhs = self._start
364        cache: defaultdict = defaultdict(list)
365        for num in range(depth):
366            yield from self.enumerate_at_depth(num, lhs, uniqueness_args, cache)
367
368    def enumerate_at_depth(
369        self,
370        depth: int,
371        lhs: Any,
372        uniqueness_args: UniquenessArgs | None = None,
373        cache: dict | None = None,
374    ) -> Generator[GrammaticalExpression, None, None]:
375        """Enumerate GrammaticalExpressions for this Grammar _at_ a fixed depth."""
376
377        if cache is None:
378            cache = defaultdict(list)
379
380        # enumerate from cache if we've seen these args before
381        args_tuple = (depth, lhs)
382        if args_tuple in cache:
383            yield from cache[args_tuple]
384        else:
385            do_unique = uniqueness_args is not None
386            if do_unique:
387                unique_dict = uniqueness_args["unique_expressions"]
388                key = uniqueness_args["key"]
389
390                def add_unique(expression: GrammaticalExpression) -> bool:
391                    """Add an expression to the unique_dict, if it is unique and shortest by the compare_func.
392                    Return the outcome boolean."""
393                    expr_key = key(expression)
394                    # if the current expression has not been generated yet
395                    # OR it is "less than" the current entry, add this one
396                    if expr_key not in unique_dict[lhs] or uniqueness_args[
397                        "compare_func"
398                    ](expression, unique_dict[lhs][expr_key]):
399                        unique_dict[lhs][expr_key] = expression
400                        return True
401                    return False
402
403            # keep a meaning -> expr dict but also depth -> expr dict
404            if depth == 0:
405                for rule in self._rules[lhs]:
406                    if rule.is_terminal():
407                        cur_expr: GrammaticalExpression = GrammaticalExpression(
408                            rule_name=rule.name, func=rule.func, children=None
409                        )
410                        if not do_unique or add_unique(cur_expr):
411                            cache[args_tuple].append(cur_expr)
412                            yield cur_expr
413            else:
414                for rule in self._rules[lhs]:
415                    # can't use terminal rules when depth > 0
416                    if rule.rhs is None:
417                        continue
418
419                    # get lists of possible depths for each child
420                    for child_depths in product(range(depth), repeat=len(rule.rhs)):
421                        if max(child_depths) < depth - 1:
422                            continue
423                        # get all possible children of the relevant depths
424                        # unique by depth?!?!
425                        children_iter = product(
426                            *[
427                                self.enumerate_at_depth(
428                                    child_depth, child_lhs, uniqueness_args, cache
429                                )
430                                for child_depth, child_lhs in zip(
431                                    child_depths, rule.rhs
432                                )
433                            ]
434                        )
435                        for children in children_iter:
436                            cur_expr = GrammaticalExpression(
437                                rule_name=rule.name, func=rule.func, children=children
438                            )
439                            if not do_unique or add_unique(cur_expr):
440                                cache[args_tuple].append(cur_expr)
441                                yield cur_expr

443    def get_unique_expressions(
444        self,
445        depth: int,
446        unique_key: Callable[[GrammaticalExpression], Any],
447        compare_func: Callable[[GrammaticalExpression, GrammaticalExpression], bool],
448        lhs: Any = None,
449        max_size: float = float("inf"),
450    ) -> dict[Any, GrammaticalExpression]:
451        """Get all unique GrammaticalExpressions, up to a certain depth, with a user-specified criterion
452        of uniqueness, and a specified comparison function for determining which Expression to save when there's a clash.
453        This can be used, for instance, to measure the minimum description length of some
454        Meanings, by using expression.evaluate(), which produces a Meaning for an Expression, as the
455        key for determining uniqueness, and length of the expression as comparison.
456
457        This is a wrapper around `enumerate`, but which produces the dictionary of key->Expression entries
458        and returns it.  (`enumerate` is a generator with side effects).
459
460        For Args, see the docstring for `enumerate`.
461
462        Note: if you additionally want to store _all_ expressions, and not just the unique ones, you should
463        directly use `enumerate`.
464
465        Returns:
466            dictionary of {key: GrammaticalExpression}, where the keys are generated by `unique_key`
467            The GrammticalExpression which is the value will be the one that is minimum among
468            `compare_func` amongst all Expressions up to `depth` which share the same key
469        """
470        unique_dict: dict[Any, dict[Any, GrammaticalExpression]] = defaultdict(dict)
471        uniqueness_args: UniquenessArgs = {
472            "unique_expressions": unique_dict,
473            "key": unique_key,
474            "compare_func": compare_func,
475        }
476        if lhs is None:
477            lhs = self._start
478        # run through generator, each iteration will update unique_dict
479        for _ in self.enumerate(
480            depth,
481            lhs=lhs,
482            uniqueness_args=uniqueness_args,
483        ):
484            if len(unique_dict) == max_size:
485                break
486            pass
487        return unique_dict[lhs]

499    @classmethod
500    def from_yaml(cls, filename: str):
501        """Read a grammar specified in a simple YAML format.
502
503        Expected format:
504
505        ```
506        start: bool
507        rules:
508        - lhs: bool
509          rhs:
510          - bool
511          - bool
512          name: "and"
513          func: "lambda p1, p2 : p1 and p2"
514        - lhs: bool
515          rhs:
516          - bool
517          - bool
518          name: "or"
519          func: "lambda p1, p2 : p1 or p2"
520        ```
521
522        Note that for each fule, the value for `func` will be passed to
523        `eval`, so be careful!
524
525        Arguments:
526            filename: file containing a grammar in the above format
527        """
528        with open(filename, "r") as f:
529            grammar_dict = load(f, Loader=Loader)
530        grammar = cls(grammar_dict["start"])
531        for rule_dict in grammar_dict["rules"]:
532            if "func" in rule_dict:
533                # TODO: look-up functions from a registry as well?
534                rule_dict["func"] = eval(rule_dict["func"])
535            if "weight" in rule_dict:
536                rule_dict["weight"] = float(rule_dict["weight"])
537            grammar.add_rule(Rule(**rule_dict))
538        return grammar

540    @classmethod
541    def from_module(cls, module_name: str) -> "Grammar":
542        """Read a grammar from a module.
543
544        The module should have a list of type-annotated method definitions, each of which will correspond to one Rule in the new Grammar.
545        See the docstring for `Rule.from_callable` for more information on how that step works.
546
547        The start symbol of the grammar can either be specified by `start = XXX` somewhere in the module,
548        or will default to the LHS of the first rule in the module (aka the return type annotation of the first method definition).
549
550        Arguments:
551            module_name: name of the module
552        """
553        module = import_module(module_name)
554        grammar = cls(None)
555        for name, value in inspect.getmembers(module):
556            # functions become rules
557            if inspect.isfunction(value):
558                grammar.add_rule(Rule.from_callable(value))
559        # set start symbol if module specifies it
560        if hasattr(module, "start") and module.start in grammar._rules:
561            grammar._start = module.start
562        # otherwise, LHS of the first rule in the module
563        else:
564            first_rule_name = next(iter(grammar._rules_by_name))
565            grammar._start = grammar._rules_by_name[first_rule_name].lhs
566        return grammar

278            re.escape(opener),
279            re.escape(closer),
280            re.escape(delimiter),

300                        children=tuple(),
